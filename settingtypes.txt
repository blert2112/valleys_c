#    ABOUT SETTINGS

# Glow adds glowing plants to the game, as well as glowing
# patches of riverbed.
valc_glow (Glowing Plants) bool false

# Add an occasional, simple building
valc_houses (Houses) bool false

# If leaf decay is disabled, leaves will not disappear, even
# when a tree is cut down.
valc_no_leaf_decay (No Leaf Decay) bool false

# Set this to true only if you have compiled the gennotify
# patch for huge caves into your mintest program.
valc_use_gennotify (Use gennotify) bool false
